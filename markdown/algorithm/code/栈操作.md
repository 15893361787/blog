### 有效括号
<!-- 给定一个只包括 '('，')'，'{'，'}'，'['，']' 的字符串，判断字符串是否有效。 -->
```js
var isValid = function(s) {
    const stack = []
    const map = {
        '(':')',
        '{':'}',
        '[':']'
    }
    let isValid = true;

    for(let i = 0;i < s.length;i++){
        if(s[i] in map){
            stack.unshift(s[i])
            continue;
        }
        
        if(!stack.length || s[i] !== map[stack.shift()]){
            isValid = false
            break;
        }

    }
    if(isValid){
        isValid = stack.length > 0?false:true
    }
    return isValid
};
```
